/*	NOTE:  These SQL Statements create the tables and insert values for the Movies database. */

/********************************************************************************/

CREATE  TABLE MOVIES(
	MovieID	        Int	              NOT NULL IDENTITY (1,1),
	Title           	Char(25)	      NOT NULL,
	Rating	        Char(10)	      NULL,
	CONSTRAINT 	MOVIE_PK 	      PRIMARY KEY(MovieID)
	);


CREATE  TABLE OWNERS(
	OwnerID	 Int   	             NOT NULL IDENTITY (10,10),
	FName	         Char(25)	 NULL,
	LName	         Char(25)	  NULL,	
	CONSTRAINT 	 OWNERS_PK 	     PRIMARY KEY(OwnerID)
	);


CREATE  TABLE LOCATION(
	LocationID	 Int   	             NOT NULL IDENTITY (100,100),
	City	         Char(25)	 NULL,
	State	         Char(2)	             NULL,
	CONSTRAINT 	 LOCATION_PK 	     PRIMARY KEY(LocationID)
	);


CREATE  TABLE THEATERS(
	TheaterID        Int 		     NOT NULL IDENTITY (1000,1000),
	TheaterName      Char(25)            NOT NULL,
	OwnerID	Int			NOT NULL,
	LocationID 	  Int  	              NOT NULL,
	CONSTRAINT 	 THEATERS_PK 	     PRIMARY KEY(TheaterID),
	CONSTRAINT	THEATERS_OWNER_FK FOREIGN KEY(OwnerID)
		REFERENCES OWNERS(OwnerID),
	CONSTRAINT 	 THEATERS_LOCATION_FK FOREIGN KEY(LocationID)
                         REFERENCES LOCATION(LocationID)
	);

CREATE  TABLE REWARDS(
	RewardID         Int	              NOT NULL IDENTITY(9000,5),
	CustFName	  Char(25)	              NOT NULL,
	CustLName	  Char(25)		NOT NULL,
	Email		  VarChar(100)           NOT NULL,
	CONSTRAINT 	  REWARDS_PK	      PRIMARY KEY (RewardID)
	);


CREATE  TABLE TICKETS(
	MovieID 	  Int  	              NOT NULL,
	TheaterID         Int	              NOT NULL,
	RewardID	  Int		NOT NULL,
	SalesDate	  DateTime            NOT NULL,
	TicketsSold       Int	    	      NOT NULL,
	CONSTRAINT 	  TICKETS_PK	      PRIMARY KEY (TheaterID, MovieID,RewardID),
	CONSTRAINT 	  TICKETS_THEATERS_FK FOREIGN KEY  (TheaterID)
			  REFERENCES THEATERS(TheaterID),
	CONSTRAINT 	  TICKETS_MOVIES_FK FOREIGN KEY  (MovieID)
			  REFERENCES MOVIES(MovieID)
			       	ON UPDATE NO ACTION
				ON DELETE NO ACTION,
	CONSTRAINT 	  TICKETS_REWARDS_FK FOREIGN KEY  (RewardID)
			  REFERENCES REWARDS(RewardID)		         	
 	);
